cmake_minimum_required(VERSION 3.10)

# Project name
project(RootFindingMethods CXX)

# Require C++17
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# --- Executables ---
add_executable(FalsePosition FalsePosition.cpp)
add_executable(NewtonRaphson NewtonRaphson.cpp)
add_executable(RootFailureFinding RootFailureFinding.cpp)

# --- Testing support ---
enable_testing()
add_test(NAME FalsePositionTest COMMAND FalsePosition)
add_test(NAME NewtonRaphsonTest COMMAND NewtonRaphson)
add_test(NAME RootFailureFindingTest COMMAND RootFailureFinding)
